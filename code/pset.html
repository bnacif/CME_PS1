# 3. Função robusta para limpeza de valores monetários
function clean_currency(val)
    if ismissing(val) || val == ""
        return 0.0
    elseif val isa Number  #isa: built-in function/operator to determine whether a value is  of a given type
        return Float64(val)
    else
        # Remove "R$" e pontos de milhar, depois troca a vírgula decimal por ponto
        s = replace(string(val), r"R\$|\." => "", "," => ".")
        return parse(Float64, s)
    end
end
